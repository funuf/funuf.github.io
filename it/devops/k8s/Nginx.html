# 使用Kubernetes部署Nginx

在/usr/local/kubernetes/nginx目录下

# deployment
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx
spec:
  selector:
    matchLabels:
      # 容器标签的名字，发布 Service 时，selector 需要和这里对应
      app: nginx
  replicas: 1
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:1.17.8
        imagePullPolicy: IfNotPresent
        ports:
        - containerPort: 80
        # 数据挂载
        volumeMounts:
        - mountPath: /usr/share/nginx/html
          name: nginx-data
          subPath: html
        - mountPath: /var/log/nginx
          name: nginx-data
          subPath: log
        - mountPath: /etc/nginx/conf.d
          name: nginx-data
          subPath: etc/conf.d
        - mountPath: /etc/nginx/nginx.conf
          name: nginx-etc
          subPath: nginx.conf
      # 数据卷配置
      volumes:
      - name: nginx-data
        persistentVolumeClaim:
         claimName: nfs-pvc-nginx
      - name: nginx-etc           
        configMap:
         name: nginx-config        
         items:
          - key: nginx.conf       
            path: nginx.conf

---
# 服务
apiVersion: v1
kind: Service
metadata:
  name: nginx
spec:
  ports:
    - port: 80
      targetPort: 80
  # ClusterIP, NodePort, LoadBalancer
  type: ClusterIP
  selector:
    app: nginx
    
---
# pv配置
apiVersion: v1
kind: PersistentVolume
metadata:
  name: nfs-pv-nginx
spec:
  # 设置容量
  capacity:
    storage: 1Gi
  # 访问模式
  accessModes:
    # 该卷能够以读写模式被多个节点同时加载
    - ReadWriteMany
  # 回收策略
  persistentVolumeReclaimPolicy: Retain
  #storageClassName: nfs-storageclass-provisioner
  volumeMode: Filesystem
  nfs:
    # NFS 服务端配置的路径
    path: "/usr/local/kubernetes/volumes/nginx"
    # NFS 服务端地址
    server: 192.168.3.130
    readOnly: false


---
# pvc 配置
apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: nfs-pvc-nginx
spec:
  accessModes:
    - ReadWriteMany
  resources:
    requests:
      storage: 1Gi
  #storageClassName: nfs-storageclass-provisioner
  volumeMode: Filesystem
  volumeName: nfs-pv-nginx

---
# configMap 配置
apiVersion: v1
kind: ConfigMap
metadata:
  name: nginx-config
data:
  # 这里是键值对数据
  nginx.conf: |
    user  nginx;
    worker_processes  1;

    error_log  /var/log/nginx/error.log warn;
    pid        /var/run/nginx.pid;


    events {
        worker_connections  1024;
    }


    http {
        include       /etc/nginx/mime.types;
        default_type  application/octet-stream;

        log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                          '$status $body_bytes_sent "$http_referer" '
                          '"$http_user_agent" "$http_x_forwarded_for"';

        access_log  /var/log/nginx/access.log  main;

        sendfile        on;
        #tcp_nopush     on;

        keepalive_timeout  65;

        #gzip  on;

        include /etc/nginx/conf.d/*.conf;
    }
