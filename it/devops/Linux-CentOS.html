# 网络配置

# 配置IP地址

# 编辑文件 "/etc/sysconfig/network-scripts/ifcfg-(网卡名称)"，根据情况修改文件内容 
# 网络类型为以太网
TYPE = Ethernet
# 手动分配ip 地址 使用静态ip地址 取值 static（静态），dhcp（动态） none （不指定）
BOOTPROTO=static
# 网卡设备名，设备名一定要跟文件名一致（linux 中设备也是一个文件）
DEVICE=ens33
# 该网卡是否随网络服务启动
ONBOOT=yes
# ip地址 
IPADDR=192.168.2.250
# 子网掩码
NETMASK=255.255.255.0
# 网关
GATEWAY=192.168.2.1
# DNS （这里使用的是阿里的DNS）
DNS1=223.5.5.5
DNS2=223.6.6.6

# 配置网络工作

# 编辑"/etc/sysconfig/network"
NETWORKING=yes

# 配置公共DNS

# 编辑文件 "/etc/resolv.conf"，根据情况修改文件内容。
nameserver 223.5.5.5
nameserver 223.6.6.6

# 重启网络服务

service network restart

# 关闭防火墙

systemctl status firewalld.service # aciive（running）说明开启
systemctl stop firewalld.service
systemctl status firewalld.service # aciive（dead）说明关闭
systemctl disable firewalld.service # 禁止防火墙服务器

# 关闭交换空间

# 关闭swap分区

swapoff -a

# 永久关闭

echo "vm.swappiness = 0">> /etc/sysctl.conf （尽量不使用交换分区，注意不是禁用）

注释掉 /etc/fstab 关于 swap 的配置

# 校对时间

# 编辑文件 "/etc/ntp.conf"，根据情况修改文件内容为：
driftfile  /var/lib/ntp/drift
pidfile   /var/run/ntpd.pid
logfile /var/log/ntp.log
restrict    default kod nomodify notrap nopeer noquery
restrict -6 default kod nomodify notrap nopeer noquery
restrict 127.0.0.1
server 127.127.1.0
fudge  127.127.1.0 stratum 10
server ntp.aliyun.com iburst minpoll 4 maxpoll 10
restrict ntp.aliyun.com nomodify notrap nopeer noquery

使用 ntpdate 的方式:

  1. yum install ntpdate
  2. ntpdate ntp.aliyun.com
  3. hwclock -w 写入硬件，避免重启失效

# yum

# 查看、卸载、安装

# 查看已安装
yum list installed | grep kube

# 卸载
yum remove kubelet kubeadm kubectl

# 安装
yum install kubelet-1.18.2 kubeadm-1.18.2 kubectl -1.18.2

# 修改yum数据源

  1. 备份 mv /etc/yum.repos.d/CentOS-Base.repo /etc/yum.repos.d/CentOS-Base.repo.backup
  2. 下载新的 CentOS-Base.repo 到 /etc/yum.repos.d/

注意：运行 yum makecache 生成缓存

# CentOS 7

wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo
# 或者
curl -o /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo

# NFS网络存储

基于网络实现对Linux目录进行挂载。

# NFS服务器

# 安装

执行以下命令安装 nfs 服务器所需的软件包

yum install -y nfs-utils

# 目录配置

执行命令 vim /etc/exports，创建 exports 文件，文件内容如下：

/usr/local/kubernetes/volumes *(rw,sync,no_subtree_check,no_root_squash)

#- /usr/local/kubernetes/volumes： 作为服务目录向客户端开放
#- *表示任何 IP 都可以访问
#- rw： 读写权限
#- sync： 同步权限
#- no_subtree_check： 表示如果输出目录是一个子目录，NFS 服务器不检查其父目录的权限
#- no_root_squash： 客户端连接服务端时如果使用的是 root，那么也拥有对服务端分享的目录的 root 权限

# 重启服务

使配置生效

systemctl enable rpcbind
systemctl enable nfs-server

systemctl start rpcbind
systemctl start nfs-server
exportfs -r

# 检查配置是否生效

exportfs
# 输出结果如下所示
/usr/local/kubernetes/volumes
		<world>

# NFS客户端

# 安装

执行以下命令安装 nfs 客户端所需的软件包

yum install -y nfs-utils

# 查看NFS服务端

执行以下命令检查 nfs 服务器端是否有设置共享目录

# showmount -e $(nfs服务器的IP)
showmount -e 172.17.216.82
# 输出结果如下所示
Export list for 192.168.3.130:
/usr/local/kubernetes/volumes *

# 进行挂载

执行挂载并且测试

mkdir /usr/local/kubernetes/nfsmount
# mount -t nfs $(nfs服务器的IP):/usr/local/kubernetes/volumes /usr/local/kubernetes/nfsmount
# mount device dir
mount -t nfs 192.168.3.130:/usr/local/kubernetes/volumes /usr/local/kubernetes/nfsmount

# 挂载完之后可以 df -h 查看
192.168.3.130:/usr/local/kubernetes/volumes   36G  2.2G   33G    7% /usr/local/kubernetes/nfsmount

# 写入一个测试文件
echo "hello nfs server" > /usr/local/kubernetes/nfsmount/test.txt
# 去往 nfs 服务器查看文件是否存在

# 解除挂载
# umount (挂载目录)
umount /usr/local/kubernetes/nfsmount/
